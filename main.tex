\section{Movement}
\coe{\keys{\arrowkeyleft} / \keys{\arrowkeyright}}{Move forward / backward in lists}
\coe{\keys{\arrowkeyup} \keys{K} / \keys{\arrowkeydown} \keys{J}}{Move up / down in cards}
\coe{\keys{B}}{Open boards menu}
\coe{\keys{/}}{Focus Search Box}

\section{Card management}
\coe{\keys{N}}{Insert New Card}
\coe{\keys{T}}{Edit Title}
\coe{\keys{E}}{Edit card description}
\coe{\keys{\return}}{Open card}
\coe{\keys{<} / \keys{>}}{Move Card to previous / next List}
\coe{\keys{C}}{Archive card}
\coe{\keys{D}}{Due date}

\section{Members}
\coe{\keys{M}}{Add / Remove Members}
\coe{\keys{S}}{Subscribe}
\coe{\keys{space}}{Assign Self}
\coe{\keys{V}}{Vote}

\section{Filters}
\coe{\keys{F}}{Open Card Filter Menu}
\coe{\keys{X}}{Clear All Filters}
\coe{\keys{Q}}{My Cards Filter}
